复制方案
import Clipboard from 'clipboard';
<button class="btn" onClick={()=>this.copy()}></button>
copy = () =>{
  const { res } = this.state;
  let clipboard = new Clipboard('.btn', {
      text: () => res
  });
  clipboard.on('success', function (e) {
      alert('复制成功')
      clipboard.destroy()
  })
}
